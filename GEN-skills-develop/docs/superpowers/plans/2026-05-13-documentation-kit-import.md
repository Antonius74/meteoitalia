# Documentation Kit Import Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Import the `feat/ark` documentation kits into the new TypeScript `nd-gen-skills` architecture as slim contract and utility packages with dependency-aware install and remove behavior.

**Architecture:** Extend the package manifest and lockfile contracts to express utility dependencies, then add a utility dependency resolver shared by `add-skill`, `remove-skill`, `sync`, and `validate`. Import documentation content as new `packages/contract/documentation-core` and `packages/utility/documentation-*` packages, keeping old public names while trimming old schema-heavy reference assets.

**Tech Stack:** TypeScript, Node.js filesystem APIs, Zod, YAML, tar, Vitest, existing `nd-gen-skills` registry builder and CLI command modules.

---

## Scope Check

This plan implements one cohesive subsystem: dependency-aware utility packages plus the documentation package import that uses that dependency model. It deliberately does not implement variant overlays, variant-specific support skills, schema-backed documentation validation, remote registry behavior, or explicit contract removal.

- [ ] **Step 2: Create dependency resolver module**

Create `src/commands/utility-dependencies.ts` with this structure:

```ts
import type { ContractManifest, UtilityManifest } from "../schemas/manifests.js";
import type { LoadedPackage } from "./install-support.js";
import { loadContractPackage, loadUtilityPackage } from "./install-support.js";

export interface ResolvedUtility {
  package: LoadedPackage<UtilityManifest>;
  requested: boolean;
  requiredBy: string[];
}

export interface UtilityDependencyClosure {
  utilities: ResolvedUtility[];
  contracts: Array<LoadedPackage<ContractManifest>>;
}

export async function resolveUtilityDependencyClosure(input: {
  registry?: string;
  requestedUtilities: string[];
}): Promise<UtilityDependencyClosure> {
  const utilities = new Map<string, ResolvedUtility>();
  const contractNames = new Set<string>();
  const visiting = new Set<string>();

  for (const utilityName of input.requestedUtilities) {
    await visitUtility({
      registry: input.registry,
      utilityName,
      requested: true,
      requiredBy: undefined,
      utilities,
      contractNames,
      visiting,
    });
  }

  const contracts = await Promise.all(
    [...contractNames].sort().map((contractName) =>
      loadContractPackage({ registry: input.registry, name: contractName }),
    ),
  );

  return {
    utilities: [...utilities.values()],
    contracts,
  };
}

async function visitUtility(input: {
  registry?: string;
  utilityName: string;
  requested: boolean;
  requiredBy: string | undefined;
  utilities: Map<string, ResolvedUtility>;
  contractNames: Set<string>;
  visiting: Set<string>;
}): Promise<void> {
  if (input.visiting.has(input.utilityName)) {
    throw new Error(`Circular utility dependency detected: ${[...input.visiting, input.utilityName].join(" -> ")}`);
  }

  const existing = input.utilities.get(input.utilityName);
  if (existing) {
    existing.requested = existing.requested || input.requested;
    if (input.requiredBy && !existing.requiredBy.includes(input.requiredBy)) {
      existing.requiredBy.push(input.requiredBy);
      existing.requiredBy.sort();
    }
    return;
  }

  input.visiting.add(input.utilityName);
  const utilityPackage = await loadUtilityPackage({ registry: input.registry, name: input.utilityName });
  const resolved: ResolvedUtility = {
    package: utilityPackage,
    requested: input.requested,
    requiredBy: input.requiredBy ? [input.requiredBy] : [],
  };
  input.utilities.set(input.utilityName, resolved);

  for (const contractName of utilityPackage.manifest.requiresContracts) {
    input.contractNames.add(contractName);
  }

  for (const dependencyName of utilityPackage.manifest.requiresUtilities) {
    await visitUtility({
      registry: input.registry,
      utilityName: dependencyName,
      requested: false,
      requiredBy: utilityPackage.manifest.name,
      utilities: input.utilities,
      contractNames: input.contractNames,
      visiting: input.visiting,
    });
  }

  input.visiting.delete(input.utilityName);
}
```

export async function buildAddUtilitiesDesiredState(input: {
  root: string;
  existingLockfile: Lockfile;
  contracts: Array<LoadedPackage<ContractManifest>>;
  utilities: ResolvedUtility[];
}): Promise<DesiredState> {
  const desiredFiles = await readExistingManagedFiles(input.root, input.existingLockfile);
  const adapter = getToolAdapter(input.existingLockfile.tool);
  const existingContractNames = new Set(input.existingLockfile.contracts.map((contract) => contract.name));
  const contractEntries = [...input.existingLockfile.contracts];
  const managedSkills = [...input.existingLockfile.managedSkills];

  for (const contract of input.contracts) {
    if (existingContractNames.has(contract.manifest.name)) {
      continue;
    }
    addPackageFiles(desiredFiles, adapter.skillDir(contract.manifest.skill.name), contract.files, contract.manifest.skill.source);
    contractEntries.push({ name: contract.manifest.name, version: contract.manifest.version });
    managedSkills.push({
      name: contract.manifest.skill.name,
      role: "contract",
      package: `contract/${contract.manifest.name}`,
    });
  }

  for (const utility of input.utilities) {
    addPackageFiles(
      desiredFiles,
      adapter.skillDir(utility.package.manifest.skill.name),
      utility.package.files,
      utility.package.manifest.skill.source,
    );
  }

  const utilityEntries = input.utilities.map((utility) => ({
    name: utility.package.manifest.name,
    version: utility.package.manifest.version,
    requested: utility.requested,
    requiredBy: utility.requiredBy,
  }));

  for (const utility of input.utilities) {
    const packageKey = `utility/${utility.package.manifest.name}`;
    if (!managedSkills.some((skill) => skill.package === packageKey)) {
      managedSkills.push({
        name: utility.package.manifest.skill.name,
        role: "utility",
        package: packageKey,
      });
    }
  }

  return {
    files: desiredFiles,
    managedSkills,
    lockfile: {
      ...input.existingLockfile,
      contracts: contractEntries,
      utilities: utilityEntries,
      managedSkills,
      managedFiles: input.existingLockfile.managedFiles,
    },
  };
}

#!/usr/bin/env bash
set -euo pipefail

ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
FAILURES=0

required_paths=(
  "SKILL.md"
  "templates/readme/repo.template.md"
  "templates/readme/local.template.md"
  "templates/architecture/repo.template.md"
  "templates/workflow/repo-workflow.template.md"
  "templates/workflow/local-workflow.template.md"
  "templates/design/repo.template.md"
  "templates/design/local.template.md"
  "templates/plan/plan.template.md"
  "templates/plan/execution-plan.template.md"
)

for rel in "${required_paths[@]}"; do
  if [ ! -e "$ROOT/$rel" ]; then
    echo "MISSING: $rel"
    FAILURES=$((FAILURES + 1))
  fi
done

if grep -R "schemas/\\|standards/\\|partials/\\|{{shared_root}}" "$ROOT" >/tmp/documentation-core-invalid-links.txt; then
  cat /tmp/documentation-core-invalid-links.txt
  FAILURES=$((FAILURES + 1))
fi

if [ "$FAILURES" -ne 0 ]; then
  echo "documentation-core validation failed with $FAILURES failure(s)."
  exit 1
fi

echo "documentation-core validation passed."
